library(tidyverse)
library(ggbeeswarm)
library(ggrepel)

big10 <- read_csv("Big10players.csv")
player <- read_csv("playeroftheyear.csv")

nebraska <- big10 |> 
  filter(Team == "Nebraska")


set.seed(1234)

ggplot() + 
  geom_beeswarm(data = big10, aes(x = Year, y = PCT), color = "grey") +
  geom_beeswarm(data = nebraska, aes(x = Year, y = PCT), color = "#e41c38") +
  geom_beeswarm(data = player, aes(x = Year, y = PCT), color = "black") +
  geom_text_repel(
    data = player, 
    aes(x = Year, y = PCT, label = Name)
